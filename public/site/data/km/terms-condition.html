កាលបរិច្ឆេទមានប្រសិទ្ធភាព៖ ២០២៥-១០-២៣

ក្រុមហ៊ុន៖ វីនថេកសូហ្វវែរឌីវេឡុបមិន (ខេមបូឌា) Co., LTD ដែលបានរៀបចំតាមច្បាប់នៃប្រទេសកម្ពុជា មានការិយាល័យចុះបញ្ជីនៅ ផ្ទះលេខ ១៣៦៣ ផ្លូវជាតិលេខ ២ តួលរកៈ ចាក់អង្រែក្រោម ខណ្ឌមានជ័យ ភ្នំពេញ កម្ពុជា។

អតិថិជន៖ អ្នក ឬអង្គភាពណាមួយ ដែលទិញ ឬប្រើប្រាស់ សេវាកម្ម (“Client”, “អ្នក”, “របស់អ្នក”)។

លក្ខខណ្ឌ និងការកំណត់ទាំងនេះ (“Terms”) គ្រប់គ្រងការចូលប្រើ និងការប្រើប្រាស់គេហទំព័រ ផ្នែកទន់ និងសេវាកម្មវិជ្ជាជីវៈ/សេវាដឹកនាំគ្រប់គ្រង (សរុបហៅថា “សេវាកម្ម”) របស់យើង។ ប្រសិនបើមានកិច្ចព្រមព្រៀង ឬពាក្យបញ្ជា (Order Form / SOW) ដែលបានចុះហត្ថលេខាដោយភាគីទាំងពីរ ហើយយោងទៅលើលក្ខខណ្ឌទាំងនេះ នោះឯកសារនោះ (រួមជាមួយលក្ខខណ្ឌទាំងនេះ) ត្រូវបានហៅថា “កិច្ចព្រមព្រៀង” (“Agreement”)។

បើអ្នកមិនយល់ព្រមនឹងលក្ខខណ្ឌទាំងនេះ សូមកុំប្រើប្រាស់សេវាកម្ម។

១) និយមន័យ (Definitions)

  • Order (ពាក្យបញ្ជា): សន្លឹកបញ្ជា Statement of Work (“SOW”) សំណើ តម្លៃ ឬឯកសារដូចគ្នាដែលបានអនុម័តដោយភាគីទាំងពីរ បញ្ជាក់សេវាកម្ម តម្លៃ និងលក្ខខណ្ឌពាក់ព័ន្ធ។
  • Software (ផ្នែកទន់): ផ្នែកទន់របស់ក្រុមហ៊ុន ឬភាគីទីបី ដែលយើងផ្តល់ ឬអនុញ្ញាតឲ្យប្រើ។
  • Client Data (ទិន្នន័យអតិថិជន): ទិន្នន័យដែលអតិថិជនផ្តល់ ឬត្រូវបានប្រមូលក្នុងនាមអតិថិជន តាមសេវាកម្ម។
  • Deliverables (លទ្ធផលការងារ): លទ្ធផលការងារ បង្កើតក្រោម SOW (ឧ. កូដ ឯកសារ កំណត់រចនាសម្ព័ន្ធ)។
  • SLA: កិច្ចព្រមព្រៀងកម្រិតសេវាកម្ម។
  • AUP: គោលនយោបាយប្រើប្រាស់អនុញ្ញាតបាន។

២) វិសាលភាពសេវាកម្ម (Scope of Services)

២.១ សេវាវិជ្ជាជីវៈ: យើងអាចផ្តល់សេវាប្រឹក្សា អនុវត្ត អភិវឌ្ឍ ផ្សារភ្ជាប់ ឬសុវត្ថិភាព និងសេវាផ្សេងទៀត តាម SOW។

២.២ សេវាដឹកនាំគ្រប់គ្រង / Hosted / SaaS: យើងអាចផ្តល់សេវាហូស្ត ត្រួតពិនិត្យ និងគាំទ្រផ្នែកទន់ ឬមូលដ្ឋានហេដ្ឋារចនាសម្ព័ន្ធ។

២.៣ ផលិតផលភាគីទីបី: យើងអាចលក់ ឬភ្ជាប់ផលិតផលភាគីទីបី ដែលស្ថិតក្រោមលក្ខខណ្ឌរបស់ភាគីទីបីនោះ។

៣) ការបញ្ជាទិញ រយៈពេល និងការបន្ត (Orders, Term, Renewal)

៣.១ អនុគ្រោះ: Order នីមួយៗ បញ្ជាក់សេវាកម្ម ថ្លៃសេវា និងរយៈពេល។ ប្រសិនបើមានការប៉ះទង្គិច Order នឹងអនុគ្រោះ។

៣.២ រយៈពេល: កិច្ចព្រមព្រៀងចាប់ផ្តើមពីថ្ងៃមានប្រសិទ្ធភាព រហូតដល់បញ្ចប់។

៣.៣ បន្តស្វ័យប្រវត្តិ: សេវាជាវនឹងបន្តស្វ័យប្រវត្តិ រាល់រយៈពេលថ្មីៗ លុះត្រាតែមានការជូនដំណឹងជាមុន (៣០ ឬ ៦០ ថ្ងៃ)។

៤) ថ្លៃសេវា និងការទូទាត់ (Fees & Payment)

៤.១ ថ្លៃសេវា: អតិថិជនបង់តាម Order រួមទាំងពន្ធដែលអនុវត្ត។

៤.២ វិក្កយបត្រ: ចេញវិក្កយបត្រនៅពេលចុះកិច្ចសន្យា ឬតាមដំណាក់កាល និងត្រូវបង់ក្នុង ៣០ ថ្ងៃ។

៤.៣ ការទូទាត់យឺត: អាចគិតការប្រាក់យឺត ១.៥% ក្នុងមួយខែ និងអាចផ្អាកសេវាកម្មក្រោយ ១០ ថ្ងៃ។

៤.៤ ចំណាយ: ចំណាយសមរម្យដែលយល់ព្រមជាមុនអាចគិតបន្ថែម។

៤.៥ គ្មានការកាត់បន្ថយ: ការទូទាត់គឺមិនអាចសងវិញ។

៥) គណនី និងការប្រើប្រាស់អនុញ្ញាតបាន (Access, Accounts & Use)

៥.១ អតិថិជនទទួលខុសត្រូវចំពោះគណនីនិងសកម្មភាពអ្នកប្រើ។

៥.២ មិនត្រូវប្រើសេវាកម្មខុសច្បាប់ បញ្ចូលមេរោគ ឬរំលោភសិទ្ធិ។

៥.៣ ក្រុមហ៊ុនអាចផ្អាកសេវា ប្រសិនបើមានហានិភ័យសុវត្ថិភាព ឬមិនទូទាត់។

៦) សិទ្ធិបញ្ញា និងលទ្ធផលការងារ (Intellectual Property)

៦.១ ភាគីនីមួយៗរក្សាសិទ្ធិលើសម្ភារៈដែលមានស្រាប់។

៦.២ ក្រុមហ៊ុនផ្តល់អាជ្ញាប័ណ្ណប្រើប្រាស់សម្ភារៈរបស់ខ្លួន តែសម្រាប់រយៈពេលកិច្ចសន្យា។

៦.៣ បន្ទាប់ពីបង់ប្រាក់ពេញ អតិថិជនទទួលអាជ្ញាប័ណ្ណប្រើប្រាស់លទ្ធផលការងារ។

៦.៤ ប្រសិនបើមានប្រភពបើកចំហ អាជ្ញាប័ណ្ណនោះគ្រប់គ្រងដោយ Open Source License។

៧) ភារកិច្ចអតិថិជន (Client Responsibilities)

៧.១ ផ្តល់ការចូលប្រើ ប្រព័ន្ធ បុគ្គលិក និងទិន្នន័យទាន់ពេលវេលា។

៧.២ ទទួលខុសត្រូវលើភាពត្រឹមត្រូវ និងភាពស្របច្បាប់នៃទិន្នន័យ។

៧.៣ រក្សាសុវត្ថិភាពប្រព័ន្ធ និងជូនដំណឹងក្រុមហ៊ុនពេលមានហានិភ័យ។

៨) ការការពារទិន្នន័យ និងឯកជនភាព (Data Protection & Privacy)

៨.១ ក្រុមហ៊ុនជា “processor” អតិថិជនជា “controller” និងនឹងចុះ DPA តាមច្បាប់។

៨.២ យើងរក្សាវិធានសុវត្ថិភាពសមរម្យ។

៨.៣ បើមានការរំលោភសុវត្ថិភាព យើងនឹងជូនដំណឹងឱ្យអ្នកភ្លាមៗ។

៨.៤ ការផ្ទេរទិន្នន័យតាមសន្ធិសញ្ញា SCC ឬតំបន់ Hosting។

៩) រក្សាការសម្ងាត់ (Confidentiality)

៩.១ ព័ត៌មានសម្ងាត់មានន័យថាព័ត៌មានមិនសាធារណៈ។

៩.២ ភាគីទទួលត្រូវការពារដូចការពារព័ត៌មានផ្ទាល់ខ្លួន។

៩.៣ ព័ត៌មានសាធារណៈ ឬទទួលបានដោយស្របច្បាប់ មិនទាត់បញ្ចូលជាសម្ងាត់។

៩.៤ អាចបង្ហាញតាមច្បាប់ បើទាមទារ។

១០) ការធានា និងការបដិសេធ (Warranties & Disclaimers)

១០.១ សេវាវិជ្ជាជីវៈធានាថាប្រតិបត្តិតាមស្តង់ដារឧស្សាហកម្ម។

១០.២ សេវា SaaS ធានាតាម SLA។

១០.៣ ផលិតផលភាគីទីបី ស្ថិតក្រោមការធានារបស់ក្រុមហ៊ុនផលិត។

១០.៤ សេវាផ្តល់ “ដូចដែលមាន” ដោយគ្មានការធានាផ្សេងទៀត។

១១) ភារកិច្ចសងខូចខាត (Indemnities)

១១.១ ក្រុមហ៊ុនការពារអតិថិជន ប្រសិនបើមានពាក្យបណ្តឹងរំលោភសិទ្ធិបញ្ញា។

១១.២ អតិថិជនការពារក្រុមហ៊ុន ប្រសិនបើមានពាក្យបណ្តឹងពីទិន្នន័យ ឬការរំលោភកិច្ចព្រមព្រៀង។

១២) កំណត់ទំនួលខុសត្រូវ (Limitation of Liability)

១២.១ ក្រុមហ៊ុននីមួយៗមានកំណត់ទំនួលខុសត្រូវមិនលើសប្រាក់ដែលបានបង់ក្នុង១២ខែមុន។

១២.២ មិនទទួលខុសត្រូវចំពោះការខូចខាតបណ្តាលដោយប្រយោល។

១២.៣ អតិថិជនត្រូវរក្សាបម្រុងទុកទិន្នន័យផ្ទាល់ខ្លួន។

១៣) កម្រិតសេវា និងការគាំទ្រ (Service Level & Support)

១៣.១ SLA បញ្ជាក់កម្រិត uptime និងសេវាគាំទ្រ។

១៣.២ ម៉ោងគាំទ្រស្តង់ដារ ៩:០០–១៧:០០ ចន្ទ–សុក្រ។

១៣.៣ ថែទាំប្រព័ន្ធអាចមានការជូនដំណឹងជាមុន។

១៤) ការបញ្ចប់ (Termination)

១៤.១ អាចបញ្ចប់នៅចុងរយៈពេលបច្ចុប្បន្នដោយជូនដំណឹងជាមុន។

១៤.២ បញ្ចប់ដោយហេតុផល បើមានការរំលោភសារៈសំខាន់។

១៤.៣ ក្រោយបញ្ចប់ អតិថិជនអាចនាំចេញទិន្នន័យក្នុង៣០ថ្ងៃ។

១៥) អនុលោមតាមច្បាប់ (Compliance)

១៥.១ ភាគីទាំងពីរត្រូវអនុវត្តច្បាប់ដែលពាក់ព័ន្ធ។

១៥.២ មិនប្រើសេវាសម្រាប់ប្រព័ន្ធសារៈសំខាន់។

១៥.៣ ព្យាយាមអនុវត្តស្តង់ដារភាពអាចចូលប្រើ។

១៦) មិនជំទាស់បុគ្គលិក (Non-Solicitation)

ក្នុងរយៈពេលកិច្ចសន្យា និង៦–១២ខែបន្ទាប់ មិនអាចជំទាស់បុគ្គលិករបស់ភាគីទៀត។

១៧) ប្រកាសសាធារណៈ (Publicity)

អាចប្រើឈ្មោះ និងឡូហ្គោរបស់អតិថិជន ក្នុងបញ្ជីអតិថិជន លុះត្រាតែអ្នកបដិសេធជាលាយលក្ខណ៍អក្សរ។

១៨) ក្រុមហ៊ុនរង និងការផ្ទេរ (Subcontractors & Assignment)

យើងអាចប្រើអ្នកអនុវត្តរង ហើយនៅតែទទួលខុសត្រូវលើការងាររបស់ពួកគេ។ មិនអាចផ្ទេរកិច្ចព្រមព្រៀងដោយគ្មានការយល់ព្រម លុះត្រាតែទៅកាន់ក្រុមហ៊ុនអាណាព្យាបាល។

១៩) ការផ្លាស់ប្តូរលក្ខខណ្ឌ (Changes to Terms)

សម្រាប់សេវាជាវ យើងអាចធ្វើបច្ចុប្បន្នភាពលក្ខខណ្ឌដោយបង្ហោះកំណែថ្មី។ ការបន្តប្រើប្រាស់បន្ទាប់គឺជាការយល់ព្រម។

២០) ច្បាប់គ្រប់គ្រង និងជម្លោះ (Governing Law & Dispute)

២០.១ កិច្ចព្រមព្រៀងគ្រប់គ្រងដោយច្បាប់ [Jurisdiction]។

២០.២ តុលាការនៅ [City, Jurisdiction] មានសិទ្ធិជាពិសេស។

២០.៣ ភាគីនឹងព្យាយាមដោះស្រាយដោយសន្តិវិធី មុនពេលអាជ្ញាកណ្ដាល។

២០.៤ អាចស្នើការបញ្ជាទប់ស្កាត់ បើមានការរំលោភសិទ្ធិបញ្ញា។

២១) ការជូនដំណឹង (Notices)

  • ទៅកាន់ក្រុមហ៊ុន៖
    • អ៊ីមែល៖ info@wintech.com.kh
    • អាសយដ្ឋាន៖ ផ្ទះលេខ ១៣៦៣ ផ្លូវជាតិលេខ ២ តួលរកៈ ចាក់អង្រែក្រោម ខណ្ឌមានជ័យ ភ្នំពេញ កម្ពុជា
    • ទូរស័ព្ទ៖ +៨៥៥ ១២ ៣៤៥ ៨៥៥
  • ទៅអតិថិជន៖ ទំនាក់ទំនងដែលបញ្ជាក់ក្នុង Order។

២២) លក្ខខណ្ឌផ្សេងៗ (Miscellaneous)

២២.១ កិច្ចព្រមព្រៀងនេះជាកិច្ចព្រមព្រៀងទាំងមូល។

២២.២ លំដាប់អនុគ្រោះ៖ Order/SOW → DPA → SLA → លក្ខខណ្ឌទូទៅ → AUP។

២២.៣ ប្រសិនបើប្រការមិនអនុវត្តបាន ផ្នែកនៅសល់នៅតែមានសុពលភាព។

២២.៤ ការមិនអនុវត្តសិទ្ធិមួយ មិនត្រូវចាត់ទុកថាបោះបង់។

២២.៥ មិនទទួលខុសត្រូវបើមានហេតុក្រៅការគ្រប់គ្រង (Force Majeure)។

២២.៦ អាចចុះហត្ថលេខាឌីជីថល ឬច្បាប់ចម្លងបាន។

២៣) លក្ខខណ្ឌគេហទំព័រ (Website Terms)

២៣.១ មាតិកាគេហទំព័រត្រូវបានផ្តល់សម្រាប់ព័ត៌មានទូទៅ។

២៣.២ អ្នកអនុញ្ញាតឲ្យយើងបង្ហោះមាតិកាដែលអ្នកផ្តល់។

២៣.៣ យើងមិនទទួលខុសត្រូវចំពោះតំណភ្ជាប់ភាគីទីបី។

២៣.៤ មិនអនុញ្ញាតឲ្យរំខានប្រព័ន្ធ ឬប្រើបូត។

២៣.៥ ផ្ញើការជូនដំណឹង DMCA ទៅ copyright@2025 វីនថេកសូហ្វវែរឌីវេឡុបមិន (ខេមបូឌា) Co.,LTD

២៤) លក្ខខណ្ឌតាមប្រទេស (Country-Specific)

  • EU/UK: បញ្ចូល GDPR ឬ UK GDPR។
  • US: បន្ថែម CCPA/CPRA។
  • អូស្ត្រាលី/កាណាដា/សិង្ហបុរី/ឥណ្ឌា: បន្ថែមច្បាប់ទិន្នន័យក្នុងតំបន់។

២៥) បែបបទ Order/SOW (Appendix)

  • ឈ្មោះគម្រោង និងវិសាលភាព
  • លទ្ធផលការងារ និងដំណាក់កាល
  • សន្មត និងអាស្រ័យភាព
  • កាតព្វកិច្ចអតិថិជន
  • ពេលវេលាប្រតិបត្តិការ
  • ថ្លៃសេវា និងកាលវិភាគបង់ប្រាក់
  • ដំណើរការផ្លាស់ប្តូរ
  • លក្ខខណ្ឌទទួលស្គាល់
  • កម្រិតសេវា និងសុវត្ថិភាពទិន្នន័យ
  • បុគ្គលសំខាន់
